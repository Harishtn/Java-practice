public class Sorting 
{

	public static void main(String[] args) 
	{
		Sorting s=new Sorting();
		
		int[] ar= {10,8,5,1};
		
		s.find_biggest(ar);
    }
    private void find_biggest(int[] ar) 
	{
		for(int i=0; i<ar.length-1; i++)
	    {
	      if(ar[i]>ar[i+1])
	      {
	        int temp = ar[i];
	        ar[i] = ar[i+1];
	        ar[i+1] = temp;
	      }
	    }
	    System.out.println("Biggest is "+ ar[ar.length-1]);	
	    
	}
}
